**Apply to:** `FeaturePanel.tsx`, `Inspector.tsx`, `Timeline.tsx`, `WorkspaceShell.tsx`

```typescript
async function executeDraftCommand<T>(
  buildCommand: DraftCommandBuilder,
  pendingCommand: string,
  applyResult: DraftCommandResultApplier<T>
): Promise<void> {
  const command = buildCommand(workspaceRef.current);
  const result = await window.videoEditorCore.executeCommand<T>(command);
  setWorkspace((current) => applyResult(current, result));
}
```

```typescript
export function formatMicroseconds(duration: Microseconds | null | undefined): string {
  const totalMilliseconds = Math.max(0, Math.floor(duration / 1000));
  const milliseconds = totalMilliseconds % 1000;
  const totalSeconds = Math.floor(totalMilliseconds / 1000);
  return `${padTime(hours)}:${padTime(minutes)}:${padTime(seconds)}.${milliseconds.toString().padStart(3, "0")}`;
}
```
